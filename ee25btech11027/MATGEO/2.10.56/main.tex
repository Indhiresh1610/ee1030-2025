\textbf{Question}.  Let two non-collinear unit vectors $\hat{a}$ and $\hat{b}$ form an acute angle. A point $\vec{P}$ moves so that at any time $t$ the position vector $\overrightarrow{OP}$ (where $\vec{O}$ is the origin) is given by $\hat{\vec{a}}\cos{t} + \hat{\vec{b}}\sin{t}$. When $\vec{P}$ is farthest from origin $\vec{O}$, let $M$ be the length of $\overrightarrow{OP}$ and $\hat{\vec{u}}$ be the unit vector along $\overrightarrow{OP}$. Then,
\begin{multicols}{2} 
    \begin{enumerate}
	    \item $\hat{\vec{u}} = \frac{\hat{\vec{a}}+\hat{\vec{b}}}{|\hat{\vec{a}}+\hat{\vec{b}}|} \text{ and } M = (1+\hat{\vec{a}} \cdot \hat{\vec{b}})^{\frac{1}{2}}$
    	\item $\hat{\vec{u}} = \frac{\hat{\vec{a}}-\hat{\vec{b}}}{|\hat{\vec{a}}-\hat{\vec{b}}|} \text{ and } M = (1+\hat{\vec{a}} \cdot \hat{\vec{b}})^{\frac{1}{2}}$
    	\item $\hat{\vec{u}} = \frac{\hat{\vec{a}}+\hat{\vec{b}}}{|\hat{\vec{a}}+\hat{\vec{b}}|} \text{ and } M = (1+2\hat{\vec{a}} \cdot \hat{\vec{b}})^{\frac{1}{2}}$
    	\item $\hat{\vec{u}} = \frac{\hat{\vec{a}}-\hat{\vec{b}}}{|\hat{\vec{a}}-\hat{\vec{b}}|} \text{ and } M = (1+2\hat{\vec{a}} \cdot \hat{\vec{b}})^{\frac{1}{2}}$
    \end{enumerate}
    \end{multicols}
\textbf{Solution}:\\
Let us solve the given equation theoretically and then verify the solution computationally. \\
Given equation:
\begin{align}
     \overrightarrow{OP}=\hat{\vec{a}}\cos{t} + \hat{\vec{b}}\sin{t}
\end{align}
From given if \textbf{P} is farthest from origin , then length of $\overrightarrow{OP}$ is given as M.From this we can say  that
\begin{align}
   M=max\norm{\overrightarrow{OP}}
\end{align}
Now,
\begin{align}
    \norm{\overrightarrow{OP}}=\sqrt{\brak{\overrightarrow{OP}}^T\brak{\overrightarrow{OP}}}
\end{align}

\begin{align}
    \norm{\overrightarrow{OP}}=\sqrt{(\hat{\vec{a}}\cos{t} + \hat{\vec{b}}\sin{t})^T(\hat{\vec{a}}\cos{t} + \hat{\vec{b}}\sin{t})}
\end{align}

\begin{align}
   \norm{\overrightarrow{OP}}=\sqrt{\norm{\hat{a}}^2cos^2t+ \norm{\hat{b}}^2sin^2t+2(\hat{a})^T(\hat{b})(cost)(sint)}
\end{align}
\begin{align}
       \norm{\overrightarrow{OP}}^2=\myvec{cost & sint}\myvec{1&(\hat{a})^T(\hat{b})\\(\hat{a})^T(\hat{b})&1}\myvec{cost\\sint}
\end{align}
Let
\begin{align}
    \Vec{x}= \myvec{cost \\ sint} \;\;and\;\;\vec{A}=\myvec{1&(\hat{a})^T(\hat{b})\\(\hat{a})^T(\hat{b})&1}
\end{align}
Then, 

\begin{align}
\norm{\overrightarrow{OP}}^2=\Vec{x}^T\vec{A}\Vec{x}
  \end{align}
Now we should find the maximum value of \textbf{$x^TAx$} such that $\norm{x}=1$\\\\
By \textbf{Rayleigh-Ritz theorem}, the maximum value of the quadratic form if \textbf{x} is a unit vector will be the largest eigenvalue $(\lambda_{max})$ of the matrix A.\\
So,
\begin{align}
    max\norm{\overrightarrow{OP}}=\sqrt{\lambda_{max}}
\end{align}
Now we will calculate the Eigen value for the matrix A :
\begin{align}
    \mydet{A-\lambda I}=0
\end{align}
\begin{align}
    \mydet{\myvec{1-\lambda&(\hat{a})^T(\hat{b})\\(\hat{a})^T(\hat{b})&1-\lambda}}=0
\end{align}
\begin{align}
    (1-\lambda)^2-((\hat{a})^T(\hat{b}))^2=0
\end{align}
\begin{align}
    1-\lambda=(\hat{a})^T(\hat{b})\;\;or\;\;1-\lambda=-(\hat{a})^T(\hat{b})
\end{align}
\begin{align}
    \lambda=1+(\hat{a})^T(\hat{b})\;\;or \;\;\lambda=1-(\hat{a})^T(\hat{b})
\end{align}
It's already given that $(\hat{a})^T(\hat{b})>0 $($\hat{a}$ and $\hat{b}$ form an acute angle . so,
\begin{align}
 \lambda_{max}=1+  (\hat{a})^T(\hat{b}) 
\end{align}
From Eq.9
\begin{align}
    max\norm{\overrightarrow{OP}}=\sqrt{1+  (\hat{a})^T(\hat{b}) }
\end{align}
The above equation can be written as
\begin{align}
   max \norm{\overrightarrow{OP}}=\sqrt{1+\hat{a}.\hat{b}}
\end{align}
From Eq.2:
\begin{align}
    M=\sqrt{1+\hat{a}.\hat{b}}
\end{align}
Now let us find the value of t for which $\norm{\overrightarrow{OP}}$ is max\\
With eigenvalue equation,We'll use matrix A and largest eigenvalue $\lambda_{max}$ such that,
\begin{align}
    \myvec{A-\lambda I}x=0
\end{align}
\begin{align}
    \myvec{-(\hat{a})^T(\hat{b})&(\hat{a})^T(\hat{b})\\(\hat{a})^T(\hat{b})&-(\hat{a})^T(\hat{b})}\myvec{cost\\sint}=\myvec{0\\0}
\end{align}
By solving this we'll get 
\begin{align}
    cost=sint
\end{align}
We already know that:
\begin{align}
    sin^2t+cos^2t=1
\end{align}
So,
\begin{align}
    sint=\frac{1}{\sqrt{2}}\;\;and\;\; cost=\frac{1}{\sqrt{2}}
\end{align}
From above result
\begin{align}
    t=\frac{\pi}{4}
\end{align}

Now unit vector $\hat{u}$ along $\overrightarrow{OP}$ is given by:
\begin{align}
    \Vec{{\hat{u}}}=\frac{\overrightarrow{OP}}{\norm{\overrightarrow{OP}}}
\end{align}
\begin{align}
\Vec{\hat{u}}=\frac{\hat{\vec{a}}\cos{t} + \hat{\vec{b}}\sin{t}}{\norm{\hat{\vec{a}}\cos{t} + \hat{\vec{b}}\sin{t}}}
\end{align}
Now subtituiting the value of $t$ in above equation:
\begin{align}
    \Vec{\hat{u}}=\frac{\Vec{\hat{a}}\frac{1}{\sqrt{2}} + \Vec{\hat{b}}\frac{1}{\sqrt{2}}}{\norm{\Vec{\hat{a}}\frac{1}{\sqrt{2}} + \Vec{\hat{b}}\frac{1}{\sqrt{2}}}}
\end{align}
\begin{align}
    \Vec{\hat{u}}=\frac{\Vec{\hat{a}} + \Vec{\hat{b}}}{\norm{\Vec{\hat{a}} + \Vec{\hat{b}}}}
\end{align}
From Eq.18 and Eq.28 (a) is correct\\
